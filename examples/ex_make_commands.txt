Full Encoder (runs both image and audio on the same args):
make encode ARGS="-i examples/cat.png -oP encoded_cat.png -p examples/secret.txt -m GREEDY -d true -iP image_processing/image_encoder/encoded_cat.png -iA oxp.wav -oA ex_full_output.wav"
make encode ARGS="-i examples/cat.png -oP encoded_cat.png -p examples/secret.txt -m SELECTIVE -d true -iP image_processing/image_encoder/encoded_cat.png -iA oxp.wav -oA ex_full_output.wav"

Full Decoder (runs both image and audio on the same args):
make encode ARGS="-i examples/encoded.wav -oP ex_full_hidden.png -o ex_full_decoded.txt -m GREEDY -d TRUE"
make encode ARGS="-i examples/encoded.wav -oP ex_full_hidden.png -o ex_full_decoded.txt -m SELECTIVE -d TRUE"

make image_encode ARGS="-i examples/cat.png -o encoded_cat.png -p examples/secret.txt -m GREEDY -d TRUE"
